14// [input] -- split --> [blocks] -- map --> [mapped blocks] -- sort --> [sorted blocks] -- reduce --> [result]
15
16template<typename InputT, typename KeyT, typename MapT, typename ReduceT>
17class MapReduce {
18public:
19 using splitted_t = std::vector<InputT>;
20 using mapped_t = std::unordered_map<KeyT, MapT>;
21 using shuffled_t = std::unordered_map<KeyT, std::vector<MapT>>;
22 using reduced_t = std::vector<ReduceT>;
23
24 using map_function_t = std::function<mapped_t(splitted_t)>;
25 using reduce_function_t = std::function<reduced_t(shuffled_t)>;
26
27 MapReduce(size_t map_concurrency_size, size_t reduce_concurrency_size,
28 map_function_t mapper, reduce_function_t reducer)
29 : map_concurrency_size_(map_concurrency_size), reduce_concurrency_size_(reduce_concurrency_size),
30 mapper_(mapper), reducer_(reducer) {}
31 ~MapReduce() = default;
32
33 reduced_t Handle(std::vector<InputT>&& input) {
34 auto splitted_data_blocks = Split(std::move(input));
35 auto mapped_data_blocks = Map(std::move(splitted_data_blocks));
36 auto shuffled_data_blocks = Shuffle(std::move(mapped_data_blocks));
37 auto reduced_data = Reduce(std::move(shuffled_data_blocks));
38 return reduced_data;
39 }
40
41private:
42 std::vector<splitted_t> Split(std::vector<InputT>&& input) {
43 std::vector<splitted_t> result;
44 result.reserve(map_concurrency_size_);
45 const size_t block_size = std::ceil(static_cast<float>(input.size()) / map_concurrency_size_);
46
47 size_t i = 0;
48 while (i < input.size()) {
49 result.emplace_back();
50 auto& block = result.back();
51 block.reserve(block_size);
52 for (; i < input.size() && block.size() < block_size; i++) {
53 auto& item = input[i];
54 block.emplace_back(std::move(item));
55 }
56 }
57
58 return result;
59 }
60
61 std::vector<mapped_t> Map(std::vector<splitted_t>&& input) {
62 auto mapped_blocks = utils::HandleInParallel<splitted_t, mapped_t>(std::move(input), mapper_);
63 return mapped_blocks;
64 }
65
66 std::vector<shuffled_t> Shuffle(std::vector<mapped_t>&& input) {
67 // distribute items over flat map
68 shuffled_t items_map;
69 for (auto& block : input) {
70 for (auto& [key, value] : block) {
71 auto bucket_it = items_map.find(key);
72 if (bucket_it == items_map.end()) {
73 auto inserted = items_map.insert({std::move(key), {}});
74 bucket_it = inserted.first;
75 }
76 bucket_it->second.push_back(std::move(value));
77 }
78 }
79
80 // prepare resulting vector with size equal to the number of reduce tasks
81 // every item consists of several map keys with corresponding data blocks
82 std::vector<shuffled_t> result;
83 result.reserve(reduce_concurrency_size_);
84 for (size_t i = 0; i < reduce_concurrency_size_; i++) {
85 result.emplace_back();
86 }
87
88 // distribute key-values across reduce task data blocks
89 size_t block_number = 0;
90 for (auto& [key, bucket] : items_map) {
91 std::sort(bucket.begin(), bucket.end(), std::less<MapT>{});
92 result[block_number][std::move(key)] = std::move(bucket);
93 block_number++;
94 if (block_number >= reduce_concurrency_size_) {
95 block_number = 0;
96 }
97 }
98
99 return result;
100 }
101
102 reduced_t Reduce(std::vector<shuffled_t>&& input) {
103 auto reduced_blocks = utils::HandleInParallel<shuffled_t, reduced_t>(std::move(input), reducer_);
104
105 // merge results
106 std::vector<ReduceT> result;
107 for (auto& block : reduced_blocks) {
108 for (auto& item : block) {
109 result.push_back(std::move(item));
110 }
111 }
112 return result;
113 }
114
115 size_t map_concurrency_size_;
116 size_t reduce_concurrency_size_;
117 map_function_t mapper_;
118 reduce_function_t reducer_;
119};
